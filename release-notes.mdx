const emojiForCategory = (category) => {
  const c = String(category ?? '').toLowerCase();
  if (c.includes('breaking')) return 'âš ï¸';
  if (c.includes('feature') || c.includes('new') || c.includes('added')) return 'ðŸš€';
  if (c.includes('fix') || c.includes('bug')) return 'ðŸ›';
  return 'âœ¨';
};

const sortCategories = (a, b) => {
  const order = [
    'breaking',
    'feature',
    'new',
    'improvement',
    'enhancement',
    'bug',
    'fix',
  ];

  const idx = (k) => {
    const c = String(k ?? '').toLowerCase();
    const i = order.findIndex((o) => c.includes(o));
    return i === -1 ? 999 : i;
  };

  return idx(a) - idx(b) || String(a).localeCompare(String(b));
};
